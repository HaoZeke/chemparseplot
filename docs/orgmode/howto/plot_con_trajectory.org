#+TITLE: Plot a general readcon CON trajectory
#+STATUS: Active

* Goal

Load any CON/convel (readcon-core) — or chemfiles-supported XYZ/PDB/GRO when
the readcon build links chemfiles — and plot energy / force profiles, overlays,
and structure strips without an eOn job directory.

* Library

#+begin_src python
from chemparseplot.api import (
    load_trajectory,
    plot_con_overview,
    plot_con_overlay,
)

traj = load_trajectory("movie.con")  # also .xyz/.pdb if chemfiles-enabled
fig = plot_con_overview(
    traj,
    energy_unit="eV",
    relative=True,
    structures="endpoints",  # none|endpoints|linspace|all
    strip_renderer="ase",    # or xyzrender
)
fig.savefig("movie_profile.pdf")

# overlay several movies
from chemparseplot.api import load_con_trajectory
t1 = load_con_trajectory("run_a.con")
t2 = load_con_trajectory("run_b.con")
import matplotlib.pyplot as plt
fig, ax = plt.subplots()
plot_con_overlay([t1, t2], labels=["A", "B"], ax=ax)
#+end_src

* CLI

#+begin_src bash
